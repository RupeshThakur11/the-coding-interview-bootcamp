_Given an integer, return an integer that is the reverse ordering of numbers._

```
function reverseInt(int) {
  const reversed = int
    .toString()
    .split('')
    .reverse()
    .join('');

  // Math.sing returns
  // 1 if (int) is positive
  // -1 if is negative
  // 0 if is 0
  return parseInt(reversed) * Math.sign(int);
}
```
